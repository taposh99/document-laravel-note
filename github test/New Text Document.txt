function selectCampaign($bidRequest, $campaigns)
{
    $selectedCampaign = null;
    $highestBid = 0;

    foreach ($campaigns as $campaign) {
        // Match country, device type, and bid floor
        $device = $bidRequest['device'];
        $imp = $bidRequest['imp'][0];

        if ($campaign['country'] === $device['geo']['country'] &&
            in_array($device['make'], explode(',', $campaign['device_make'])) &&
            $campaign['price'] >= $imp['bidfloor']) {

            // Check if current campaign bid is higher
            if ($campaign['price'] > $highestBid) {
                $highestBid = $campaign['price'];
                $selectedCampaign = $campaign;
            }
        }
    }

    return $selectedCampaign;
}
